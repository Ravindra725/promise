Q6. Create examples to explain callback hell function

ans: 
function rk(callback){
    console.log("hello callback")
    callback();
}
function rk1(callback){
    console.log("hello callback again 1")
    callback();
}
function rk2(callback){
    console.log("hello callback again 2")
    callback();
}
function rk3(callback){
    console.log("hello callback again 3")
    callback();
}
function rk4(callback){
    console.log("hello callback again 4")
}

rk(()=>{
    rk1(()=>{
        rk2(()=>{
            rk3(()=>{
                rk4();
            })
        })
    })
})
OutPut: 
hello callback
hello callback again 1
hello callback again 2
hello callback again 3
hello callback again 4
